$(document).ready(function() {
$('#documents').delegate('a', 'click', function () {
//左导航菜单点击时调用增加TAB的函数
if (!$(this).attr("link")) {
return false;
} else {
addTab($(this));
}
});
$(".data_chart a").click(function() {//iframe框架内菜单点击时调用增加TAB的函数
parent.addTab($(this));
});
$('#tabs').delegate('a.tab', 'click', function () {//切换tab
// 隐藏所有tab和iframe
$("#tabs li").removeClass("current");
$("#content iframe").hide();
var this_id = $(this).attr("id");//读取首页的id
if(this_id=='index'){//判断id是否等于index，是就显示
$("#home_content").show();
}
// 显示点击的tab及iframe
$(this).parent().addClass("current");
var contentname = $(this).attr("id") + "_content";
$("#" + contentname).show();
});
$('#tabs').delegate('a.remove', 'click', function (e) {//删除TAB
// Get the tab name
var tabid = $(this).parent().find(".tab").attr("id");
// remove tab and related content
var contentname = tabid + "_content";
$("#" + contentname).remove();
$(this).parent().remove();
// if there is no current tab and if there are still tabs left, show the first one
if ($("#tabs li.current").length == 0 && $("#tabs li").length > 0) {
// find the first tab
var firsttab = $("#tabs li:first-child");
firsttab.addClass("current");
// get its link name and show related content
var firsttabid = $(firsttab).find("a.tab").attr("id");
$("#" + firsttabid + "_content").show();
}
});
});
function addTab(link) {//增加TAB
// If tab already exist in the list, return
if ($(link).attr("rel")=='home') {
$("#tabs li").removeClass("current");
$("#content iframe").hide();
$("#index").parent("li").addClass("current");
$("#home_content").show();
return false;
}
if ($("#" + $(link).attr("rel")).length != 0) {
$("#tabs li").removeClass("current");
$("#content iframe").hide();
$("#" + $(link).attr("rel")).parent("li").addClass("current");
$("#" + $(link).attr("rel") + "_content").show();
return
}
// hide other tabs
$("#tabs li").removeClass("current");
$("#content iframe").hide();
tabs_num = $("#tabs li").length
if(tabs_num > 7){
xxts()
}
var html="<li class='current'><a class='tab' id='" +
$(link).attr("rel") + "' href='#'>" + $(link).text() +
"</a><a href='#' class='remove'></a></li>";
$("#tabs").append(html);
$("#content").append("<iframe id='" + $(link).attr("rel") + "_content' name='right' frameborder='0' src='"+$(link).attr("link")+"' scrolling='auto' ></iframe>");
// set the newly added tab as current
$("#" + $(link).attr("rel") + "_content").show();
